# 第4章_7.AVL树的实现
* **AvlTree.h**
```c
#ifndef _AvlTree_H_
#define _AvlTree_H_

typedef int ElementType;
struct AvlTreeNode;
typedef struct AvlTreeNode *Position;
typedef struct AvlTreeNode *AvlTree;

AvlTree MakeEmpty(AvlTree T);
AvlTree Insert(ElementType X,AvlTree T);//插入操作应该返回根节点地址，所以返回值是树
AvlTree Delete(ElementType X,AvlTree T);//同上
Position Find(ElementType X,AvlTree T);
Position FindMax(AvlTree T);
Position FindMin(AvlTree T);
ElementType Retrieve(Position P);
#endif
```

* **AvlTree.c**
```c
//      不是标准答案，可能会有错
//      另外Delete操作会比Insert操作还要复杂，书中暂未讨论
//      如果Delete操作采用懒惰删除的话，需要注意修改查找和插入操作的判断条件
#include "AvlTree.h"
#include <stdlib.h>
struct AvlTreeNode
{
    ElementType Element;
    int Hight;
    AvlTree Left;
    AvlTree Right;
};

AvlTree MakeEmpty(AvlTree T)
{
    if (T != NULL)
    {
        MakeEmpty(T->Left);
        MakeEmpty(T->Right);
        free(T);
    }
    return T;
}
AvlTree Insert(ElementType X, AvlTree T) //插入操作应该返回根节点地址，所以返回值是树
{
    if (T==NULL)
    {
        T=malloc(sizeof(struct AvlTreeNode));
        if(T==NULL)
            FatalError("out of memory");
        T->Element=X;
        T->Hight=0;
    }else if(X<T->Element)
    {
        T->Left=Insert(X,T->Left); //插入进左子树
        if(Hight(T->Left)-Hight(T->Right)==2) //左子树高减右子树高
            if(X<T->Left->Element) //判断是否插入在了左子树的左子树
                T=singleRotateToRight(T);
            else //否则插入在了左子树的右子树
                T=doubleRotateToRight(T);
    }else if(X>T->Element)
    {
        T->Right=Insert(X,T->Right);
        if(Hight(T->Right)-Hight(T->Left)==2)
            if(X>T->Right->Element)
                T=singleRotateToLeft(T);
            else
                T=doubleRotateToLeft(T);
    }else if(X==T->Element)
        ;
    
    // 如果其最高子树是一颗空树(-1) 那么当前树的高度应该是-1+1=0
    // 如果其最高子树是一片叶子(0) 那么当前树的高度应该是0+1=1
    // 如果其最高子树有一片叶子(1) 那么当前树的高度应该是1+1=1
    T->Hight = MAX(Hight(T->Left),Hight(T->Right)) + 1;
    return T;
}
AvlTree Delete(ElementType X, AvlTree T)
{
}
Position Find(ElementType X, AvlTree T)
{
    if (T == NULL)
        return T;
    else if (X < T->Element)
        return Find(X, T->Left);
    else if (X > T->Element)
        return Find(X, T->Right);
    else
        return T;
}
Position FindMax(AvlTree T)
{
    if (T == NULL)
        return T;
    else if (T->Right == NULL)
        return T;
    else
        return FindMax(T->Right);
}
Position FindMin(AvlTree T)
{
    if (T == NULL)
        return T;
    else if (T->Left == NULL)
        return T;
    else
        return FindMin(T->Left);
}
ElementType Retrieve(Position P)
{
    return P != NULL ? P->Element : 0;
}
static int MAX(int a,int b)
{
    return a>b? a:b;
}
static int Hight(AvlTree T)
{
    if (T==NULL)
        return -1;
    else
        return T->Hight;
}

static AvlTree singleRotateToRight(AvlTree T)
{   //调整指针
    AvlTree newRoot=T->Left;
    T->Left=newRoot->Right;
    newRoot->Right=T;
    //更新高度时应该先更新子节点再更新根节点
    T->Hight=MAX(Hight(T->Left),Hight(T->Right))+1;
    newRoot->Hight=MAX(Hight(newRoot->Left),Hight(newRoot->Right))+1;
    return newRoot;
}

static AvlTree singleRotateToLeft(AvlTree T)
{   //调整指针
    AvlTree newRoot=T->Right;
    T->Right=newRoot->Left;
    newRoot->Left=T;
    //更新高度时应该先更新子节点再更新根节点
    T->Hight=MAX(Hight(T->Left),Hight(T->Right))+1;
    newRoot->Hight=MAX(Hight(newRoot->Left),Hight(newRoot->Right))+1;
    return newRoot;
}

static AvlTree doubleRotateToRight(AvlTree T)
{
    T->Left=singleRotateToLeft(T->Left);//先对根节点的左子树做右旋操作
    return singleRotateToRight(T);//再对根节点做左旋操作
}

static AvlTree doubleRotateToLeft(AvlTree T)
{
    T->Right=singleRotateToRight(T->Right);//先对根节点的左子树做左旋操作
    return singleRotateToLeft(T);//再对根节点做右旋操作
}
```
